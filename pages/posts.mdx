# Thoughts

import { getPagesUnderRoute } from 'nextra/context'
import Link from 'next/link'

export function BlogIndex({ more = 'Read more' }) {
  return getPagesUnderRoute('/posts')
    .sort((a, b) => {
      return new Date(b.frontMatter.date) > new Date(a.frontMatter.date)
        ? 1
        : -1
    })
    .map((page) => {
      return (
        <div key={page.route} className="mb-10">
          <Link
            style={{ color: 'inherit', textDecoration: 'none' }}
            className="post-link block font-semibold mt-8 text-2xl flex"
            href={page.route}
          >
            <h3 className="flex-1 m-0 post-title">
              {page.meta?.title || page.frontMatter?.title || page.name}
            </h3>
            <time className="opacity-50 text-sm date">
              {page.frontMatter.date}
            </time>
          </Link>
        </div>
      )
    })
}

<BlogIndex />
